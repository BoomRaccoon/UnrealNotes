    \section{Gamefeatures}
        \begin{itemize}
            \item offer a quick way to activate and deactivate plugins
            \item located in \code{Plugins/GameFeatures/} \textbf{on disk}
            \item located in \code{Plugins/} \textbf{in content browser}
            \item Plugins to activate:
            \begin{itemize}
                \item \code{Game Features}
                \item \code{Modular Gameplay}
            \end{itemize}
            \item automatically add GameFeatures as Primary assets(through \code{Meesage Log}) or manually:
            \begin{itemize}
                \item \code{Edit -> Project Settings -> Asset Manager}
                \item Primary Asset Types to Scan
                \item add one called GameFeatureData
                \item uses the \code{GameFeatureData} base-class
                \item add the directory \code{/Game/Unused} to the Directories array
                \item Rules and set \code{Cook Rule} = \code{Always Cook}
            \end{itemize}

        \end{itemize}
